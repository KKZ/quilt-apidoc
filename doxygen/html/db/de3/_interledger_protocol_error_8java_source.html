151  // Disallow the triggeredBy from being included in the forwardedBy. The rationale is that
152  // the triggering node should not accidentally add itself to the forwarding addresses.
153  // Likewise, if that ever happens with an incoming error, then we should throw an exception.
154  this.getForwardedByAddresses().stream()
155  .filter(interledgerAddress -> interledgerAddress.equals(this.getTriggeredByAddress()))
156  .findFirst()
157  .ifPresent(interledgerAddress -> {
158  // Throw an exception here because if this occurs, it indicates a packet loop, and
159  // we don't want to simply remove the address from the ForwardedBy list and send
160  // the packet on, because doing so would likely mean it will come back to us.
161  throw new IllegalArgumentException(String.format(
162  "TriggeredByAddress \"%s\" was found in the ForwardedByAddresses list, which "
