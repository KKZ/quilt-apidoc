168  private static Map<String, List<String>> splitQuery(String queryParams)
169  throws UnsupportedEncodingException {
170  // Used to avoid dependencies on additional libraries. Lightly adapted from
171  // http://stackoverflow.com/questions/13592236/parse-a-uri-string-into-name-value-collection
172  final Map<String, List<String>> query_pairs = new LinkedHashMap<String, List<String>>();
173  final String[] pairs = queryParams.split("&");
174  for (String pair : pairs) {
175  final int idx = pair.indexOf("=");
176  final String key = idx > 0 ? URLDecoder.decode(pair.substring(0, idx), "UTF-8") : pair;
177  if (!query_pairs.containsKey(key)) {
178  query_pairs.put(key, new LinkedList<String>());
179  }
180  final String value = idx > 0 && pair.length() > idx + 1
181  ? URLDecoder.decode(pair.substring(idx + 1), "UTF-8") : null;
182  query_pairs.get(key).add(value);
183  }
184  return query_pairs;
185  }
